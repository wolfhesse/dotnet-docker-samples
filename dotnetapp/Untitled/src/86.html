using System;
using System.IO;
using DnsLib.ComponentLibrary.Authentication;
using DnsLib.SysRes;
using RestSharp;
using Xunit;

namespace DnsLib.TestDriving.Operations
{
    public static class WooProductUnitTest
    {
        [Fact(Timeout = 2500)]
        public static void TestRcsProductFetch()
        {
            var auth = new AuthContainer();
            AseAuthVaultForTesting.PutRcs2ConsumerAuthInto(auth);

            var client = new RestClient(PlatformSysGen.UrlRcs2);
            var request = new RestRequest(PlatformSysGen.ApiEndpointProducts);


            request.AddParameter("per_page", 3);
            // als letzten schritt vor client.Execute :request
            request.BuildOAuth1QueryString(client, auth.Key, auth.Secret);

            var response = client.Execute(request);
            var content = response.Content;

            Console.Out.WriteLine(content.Substring(0, 32));

            // schreibs in eine json datei
            using (var writer = new StreamWriter(File.Open(PlatformSysGen.JsonPathDataDProducts, FileMode.Create)))
            {
                writer.Write(content);
            }
        }
    }
}
