using System.Collections.Generic;
using DnsLib.AbstractArchitecture.AseArchitecture.Definitions;
using DnsLib.SysRes;

namespace DnsLib.FloorPlaan.PkgCt.BusinessPackageDistribution
{
    public class SyncPlatform
    {
        private readonly Dictionary<string, object> _memoryBank;

        public SyncPlatform()
        {
            _memoryBank = new Dictionary<string, object>();
        }

        public Dictionary<string, object> FindAccount(
            (string sourceSystem, string targetSystem) syncTaskDescriptor)
        {
            _memoryBank[nameof(ServiceAccessPoint.MemorySlot.Done)] = false;
            // fetch sourceSystemAccount
            _memoryBank[nameof(ServiceAccessPoint.MemorySlot.FindAccountForOpCount)] = 0;


            new FindAccountForOp(syncTaskDescriptor.sourceSystem, _memoryBank)
                .Execute(ServiceAccessPoint.MemorySlot.SourceSystemAccount);

            new FindAccountForOp(syncTaskDescriptor.targetSystem, _memoryBank)
                .Execute(ServiceAccessPoint.MemorySlot.TargetSystemAccount);

            var config = (memoryBank: _memoryBank, sender: PlatformDeveloperSection.SystemUser.Rogera);

            new ServiceAccessPoint(_memoryBank).CountServiceProvision(config);
            _memoryBank[nameof(ServiceAccessPoint.MemorySlot.Done)] = true;
            return _memoryBank;
        }
    }
}
