using System;
using System.ComponentModel;

namespace DnsLib.FactoryFloor.Utility
{
    #region using directives

    #endregion

    /// <summary>The component 2.</summary>
    public partial class Component2 : Component
    {
        /// <summary>The ev processing.</summary>
        public event EventHandler EvProcessing;

        /// <summary>The ev started.</summary>
        public event EventHandler EvStarted;

        /// <summary>The ev stopped.</summary>
        public event EventHandler EvStopped;

        /// <summary>The start.</summary>
        public void Start()
        {
            OnEvStarted();

            //// processing
            Process();
            Stop();
        }

        /// <summary>The on ev processing.</summary>
        protected virtual void OnEvProcessing()
        {
            EvProcessing?.Invoke(this, EventArgs.Empty);
        }

        /// <summary>The on ev started.</summary>
        protected virtual void OnEvStarted()
        {
            EvStarted?.Invoke(this, EventArgs.Empty);
        }

        /// <summary>The on ev stopped.</summary>
        protected virtual void OnEvStopped()
        {
            EvStopped?.Invoke(this, EventArgs.Empty);
        }

        /// <summary>The process.</summary>
        private void Process()
        {
            OnEvProcessing();
        }

        /// <summary>The stop.</summary>
        private void Stop()
        {
            OnEvStopped();
        }
    }
}
