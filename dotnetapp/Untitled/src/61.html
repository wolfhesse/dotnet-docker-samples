namespace DnsLib.FactoryFloor.TodoComponent
{
    /// <summary>
    ///     The add task.
    /// </summary>
    public static class AddTodo
    {
        /// <summary>The execute.</summary>
        /// <param name="title">The title.</param>
        public static void Execute(string title)
        {
            var task = TodoBuilder.BuildTodo(title);
            TodoController.TodoRepository.Persist(task);
        }

        /// <summary>The execute.</summary>
        /// <param name="todoRepository">The task repository.</param>
        /// <param name="todoItem">The todo task.</param>
        internal static void Execute(AbstractTodoRepository todoRepository, TodoItem todoItem)
        {
            todoRepository.Persist(todoItem);
        }
    }
}
