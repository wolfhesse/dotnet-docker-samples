using System.Collections.Generic;
using DnsLib.SysRes;

namespace DnsLib.FactoryFloor.Utility
{
    public static class CodeDiary
    {
        public static readonly Dictionary<string, string> Entries = new Dictionary<string, string>();

        public static void Create(string eintragsmarker, string eintragstext)
        {
            Entries[eintragsmarker] = eintragstext;
        }

        public static void Dump()
        {
            EnvironmentManager.WriteLine("Dump of CodeDiary");
            foreach (var entry in Entries) EnvironmentManager.WriteLine(entry.Value, entry.Key);
        }
    }
}
