    /// <summary>The es exper engine.</summary>
    public static class EsExperEngine
    {
        /// <summary>The play with es tweets.</summary>
        /// <returns>
        ///     The
        ///     <see>
        ///         <cref>List</cref>
        ///     </see>
        ///     .
        /// </returns>
        public static List<TweetModel> PlayWithEsTweets()
        {
            var sampleTweet = EnvironmentManager.SampleTweet;
            var tweets = EsOperationsEngine.EsWriteAndDupTweet(sampleTweet);
            var sw = EnvironmentManager.GetResultStreamWriter();
            sw.WriteLine(new string('-', 72));
            sw.WriteLine("play w/ tweets (es)");
            sw.WriteLine(tweets[1]);
            sw.Close();
            return tweets;
        }
    }
