namespace DnsLib.FloorPlaan.PkgCt.BusinessPackageCustomer
{
    public class CreateCustomerAction
    {
        public CreateCustomerAction(CustomerRepository customerRepository, string name)
        {
            customerRepository.CreateCustomer(name);
            Status = Results.Success;
        }

        public enum Results
        {
            Success,
            Failure
        }

        public Results Status { get; set; }
    }
}
