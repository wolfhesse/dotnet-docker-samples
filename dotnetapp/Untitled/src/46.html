using System;
using System.Collections.Generic;
using DnsLib.SysRes;
using WooCommerceNET.WooCommerce.v2;

namespace DnsLib.FactoryFloor.ShopComponent.Entity
{
    #region using directives

    #endregion

    /// <summary>The product future.</summary>
    public class ProductFuture
    {
        /// <summary>The container.</summary>
        // ReSharper disable once PrivateFieldCanBeConvertedToLocalVariable
        private readonly Product _container;

        /// <summary>Initializes a new instance of the <see cref="ProductFuture" /> class.</summary>
        public ProductFuture()
        {
            _container = new Product();
            EnvironmentManager.WriteLine(_container);
        }

        public static void FnLogging(List<Product> p)
        {
            p.ForEach(
                px => { Console.Out.WriteLine($"\tproduct: {px.id}, {px.name}\n\t\t .. {px.short_description}"); });
        }

    }
}
