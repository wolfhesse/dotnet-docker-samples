using System.Collections.Generic;
using DnsLib.AbstractArchitecture.AseArchitecture.Definitions;

namespace DnsLib.SysRes
{
    public class ServiceAccessPoint
    {
        private Dictionary<string, object> _memoryBank;

        public ServiceAccessPoint(Dictionary<string, object> memoryBank)
        {
            _memoryBank = memoryBank;
        }

        public enum MemorySlot
        {
            SourceSystemAccount,
            TargetSystemAccount,
            SourceSystemAuth,
            FindAccountForOp,
            Done,
            SystemAuth,
            FindAccountForOpCount
        }

        public bool CountServiceProvision(
            (Dictionary<string, object> memoryBank, PlatformDeveloperSection.SystemUser sender) config)
        {
            _memoryBank = config.memoryBank;
            /* klammer contract
               protocol: _memoryBank[nameof( klass//obj )]
               */
            _memoryBank[nameof(CountServiceProvision)] = 0;


            var counter = 1;
            _memoryBank[nameof(CountServiceProvision)] = counter;
            return true;
        }
    }
}
