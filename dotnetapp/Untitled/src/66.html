using System;
using System.IO;
using DnsLib.SysRes;

namespace DnsLib.FactoryFloor.Utility
{
    #region using directives

    #endregion

    /// <summary>
    ///     The general operations.
    /// </summary>
    public static class GeneralOperations
    {
        /// <summary>The err_handling_bail_out.</summary>
        /// <param name="reason">The reason.</param>
        public static void ErrHandlingBailOut(string reason = null)
        {
            EnvironmentManager.WriteLine($"bailing out: rc .. {reason} // ");
        }

        /// <summary>The handle bailing out.</summary>
        /// <param name="textWriter">The text writer.</param>
        /// <param name="reason">The reason.</param>
        public static void HandleBailingOut(TextWriter textWriter, string reason = null)
        {
            textWriter.WriteLine($"bailing out: rc .. //////////////////// {reason} //////////////////// ");
        }
    }

    /// <summary>The helper.</summary>
    public class Helper
    {
        /// <summary>The fn separator 36.</summary>
        public static string FnSeparator36 => new string('_', 36);

        /// <summary>The fn out separator 36.</summary>
        /// <param name="textWriter">The text writer.</param>
        /// <param name="character">The character.</param>
        public static void FnOutSeparator36(TextWriter textWriter = null, char character = '_')
        {
            if (textWriter == null) textWriter = Console.Out;
            textWriter.WriteLine(new string(character, 36));
        }

        /// <summary>The fn out separator 72.</summary>
        /// <param name="textWriter">The text writer.</param>
        /// <param name="character">The character.</param>
        public static void FnOutSeparator72(TextWriter textWriter = null, char character = '_')
        {
            if (textWriter == null) textWriter = Console.Out;
            textWriter.WriteLine(new string(character, 72));
        }
    }
}
