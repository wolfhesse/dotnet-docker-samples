using System;
using System.Collections.Generic;
using DnsLib.AbstractArchitecture.AseArchitecture;

namespace DnsLib.FactoryFloor.Utility
{
    internal class ActorNodeBaseImpl : IVirtualActorNode
    {
        public ActorNodeBaseImpl(string name, List<IVirtualActorNode> connectionsList = null)
        {
            Name = name;
            ConnectionsList = connectionsList ?? new List<IVirtualActorNode>();
        }

        public List<IVirtualActorNode> ConnectionsList { get; set; }

        public string Name { get; }

        public void RegisterAsEventHandler(IVirtualActorNode eventSource)
        {
            ConnectionsList.Add(eventSource);
        }

        public override string ToString()
        {
            var s = base.ToString();
            s += Environment.NewLine + $"\t{Name} - NodeCount: {ConnectionsList.Count}";
            return s;
        }
    }
}
