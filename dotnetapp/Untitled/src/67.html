    /// <summary>The tweet factory.</summary>
    public static class TweetEngine
    {
        /// <summary>The ev sample tweet was requested.</summary>
        public static event EventHandler EvSampleTweetWasRequested;

        /// <summary>The build tweet.</summary>
        /// <param name="user">The user.</param>
        /// <returns>The <see cref="TweetModel" />.</returns>
        public static TweetModel BuildTweet(string user)
        {
            return new TweetModel
            {
                User = user,
                PostDateTime = DateTimeOffset.Now
            };
        }

        /// <summary>The dump tweet.</summary>
        /// <param name="obj">The obj.</param>
        public static void DumpTweet(TweetModel obj)
        {
//            Console.WriteLine(null != obj ? $"dumpTweet: {obj.Id} : {obj.User} // {obj.Value}" : "obj is null");
            // reduce output
            Console.WriteLine(null != obj ? $"dumpTweet: {obj.Id} : {obj.User}" : "obj is null");
        }

        /// <summary>The fn create tweet.</summary>
        /// <param name="message">The message.</param>
        /// <returns>The <see cref="TweetModel" />.</returns>
        public static TweetModel FnCreateTweet(string message)
        {
            var t = new TweetModel
            {
                PostDateTime = DateTimeOffset.Now.DateTime,

                // Id = tweets++,
                User = typeof(ProgramsEnvironment).ToString(),
                Value = message
            };
            return t;
        }

        /// <summary>The get sample tweet for.</summary>
        /// <param name="klass">The klass.</param>
        /// <returns>The <see cref="TweetModel" />.</returns>
        public static TweetModel GetSampleTweetFor(INeedSampleTweet klass)
        {
            // todo: EvSampleTweetReqeuestedBy..
            OnEvSampleTweetWasRequested(klass);

            var pTweet = new TweetModel
            {
                Id =
                    null, // typeof(InteropTypes.V1.TweetModel) + ".GetSampleTweetFor",
                // Id = 1,
                User = "rogeraaut",
                // todo DSS-14
                PostDateTime = DateTime.Now,
                Value =
                    $"Trying out NEST, so far so good? \n{typeof(TweetModel)}.GetSampleTweetFor @ {DateTimeOffset.Now}"
            };
            return pTweet;
        }

        /// <summary>The on ev sample tweet was requested.</summary>
        /// <param name="klass">The klass.</param>
        public static void OnEvSampleTweetWasRequested(INeedSampleTweet klass)
        {
            EvSampleTweetWasRequested?.Invoke(klass, EventArgs.Empty);
        }
    }
