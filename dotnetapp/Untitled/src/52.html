using System;
using DnsLib.SysRes;
using Xunit;


namespace DnsLib.TestDriving.Core
{
    #region using directives

    #endregion

    /// <summary>The tweet factory tests.</summary>
    public sealed class TweetEngineTests : IDisposable
    {
        /// <summary>The setup.</summary>
        public TweetEngineTests()
        {
            EnvironmentManager.AddSampleTweetWasRequestedEventHandler(ConsoleOutSampleTweetWasRequested);
        }

        public void Dispose()
        {
            EnvironmentManager.WriteLine($"{typeof(TweetEngineTests)}", "Dispose");
        }

        /// <summary>The build tweet test.</summary>
        [Fact(Timeout = 2500)]
        public static void BuildTweetTest()
        {
            Assert.Equal(EnvironmentManager.SampleTweet.User,
                TweetEngine.BuildTweet("rogeraaut").User);
            Console.Out.WriteLine("zweiter aufruf: sample tweet: = {0}", EnvironmentManager.SampleTweet);
        }

        /// <summary>The interface need sample tweet has version.</summary>
        [Fact(Timeout = 2500)]
        public static void InterfaceNeedSampleTweetHasVersion()
        {
            var t = EnvironmentManager.SampleTweet;
            Assert.NotNull(t.Version);
            var tweet = t as INeedSampleTweet;
            Assert.True(tweet != null);
            Assert.NotNull((t as INeedSampleTweet).Version);
        }


        /// <summary>The console out sample tweet was requested.</summary>
        /// <param name="sender">The sender.</param>
        /// <param name="eventArgs">The event args.</param>
        private void ConsoleOutSampleTweetWasRequested(object sender, EventArgs eventArgs)
        {
            Console.Out.WriteLine("Ev: sample tweet was requested");
        }
    }
}
