namespace DnsLib.FactoryFloor.TodoComponent
{
    /// <summary>The task builder.</summary>
    public class TodoBuilder
    {
        /// <summary>The build task.</summary>
        /// <param name="title">The title.</param>
        /// <returns>The <see cref="TodoItem" />.</returns>
        public static TodoItem BuildTodo(string title)
        {
            var t = new TodoItem(title) {ExtraCreatorInfo = typeof(TodoBuilder)};
            return t;
        }
    }
}
