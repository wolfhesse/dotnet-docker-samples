using System;
using DnsLib.FactoryFloor.ShopComponent;
using DnsLib.FactoryFloor.ShopComponent.AseWooCommerceNET;
using DnsLib.SysRes;
using WooCommerceNET.WooCommerce.v2;
using Xunit;


namespace DnsLib.TestDriving.Trainer.Cs7
{
    public class TupleTests
    {
        [Fact(Timeout = 2500)]
        public void DumpSampleProductTest()
        {
            var restApi = WooStuffAuthAdapter.RestApiDefault();
            IShopConfiguration configuration = new WooCommerceConfiguration(restApi);
            IShopAdapter adapter = new WooCommerceAdapter();
            var shopEngine = new ShopEngine(adapter, configuration);

            var product = shopEngine.AddProduct(new Product
            {
                name = $@"Product created at {DateTimeOffset.Now}",
                description = $@"Sample Product created by {this}",
                short_description = $@"<p><small>Sample Product</small> via {this}<br /></p>"
            });

            // redef stru: name to title
            (string description, string short_description, int? id, string title) copiedTuple
                = (product.description, product.short_description, product.id, product.name);
            Console.WriteLine(copiedTuple);
            Assert.False(false);
        }

        [Fact(Timeout = 2500)]
        public static void TestSimple()
        {
            var tuple = (1, 2);

            Console.WriteLine(tuple);

            (int a, int b) namedTupleSample = tuple;
            Console.Out.WriteLine("namedTupleSample = {0}", namedTupleSample);
            namedTupleSample.b++;
            Console.Out.WriteLine("namedTupleSample = {0}", namedTupleSample);

            Assert.False(false);
        }
    }
}
