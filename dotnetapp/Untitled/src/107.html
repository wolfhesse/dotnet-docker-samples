using System.Collections.Generic;
using DnsLib.ComponentLibrary.Authentication;
using DnsLib.SysRes;

namespace DnsLib.FloorPlaan.PkgCt.BusinessPackageDistribution
{
    public class FindAccountForOp
    {
        private readonly Dictionary<string, object> _memoryBank;

        public FindAccountForOp(string systemName, Dictionary<string, object> memoryBank)
        {
            _memoryBank = memoryBank;
            _memoryBank[nameof(ServiceAccessPoint.MemorySlot.FindAccountForOp)] = "inflight";

            switch (systemName)
            {
                case "rcs1":
                {
                    var c = new AuthContainer();
                    AseAuthVaultForTesting.PutRcs1ConsumerAuthInto(c);
                    _memoryBank[nameof(FindAccountForOp)] = c;
                    break;
                }
                case "rcs2":
                {
                    var c = new AuthContainer();
                    AseAuthVaultForTesting.PutRcs2ConsumerAuthInto(c);
                    _memoryBank[nameof(FindAccountForOp)] = c;
                    break;
                }

                default:
                {
                    _memoryBank[typeof(FindAccountForOp).ToString()] = false;
                    break;
                }
            }
        }

        public void Execute(ServiceAccessPoint.MemorySlot slot)
        {
            int i = int.Parse(_memoryBank[nameof(ServiceAccessPoint.MemorySlot.FindAccountForOpCount)].ToString()) + 1;
            _memoryBank[nameof(ServiceAccessPoint.MemorySlot.FindAccountForOpCount)] = i;

            _memoryBank[slot.ToString()] = _memoryBank[nameof(ServiceAccessPoint.MemorySlot.FindAccountForOp)];
            _memoryBank[nameof(ServiceAccessPoint.MemorySlot.FindAccountForOp)] = "executed";
        }
    }
}
