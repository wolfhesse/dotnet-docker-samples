using System;
using System.Collections.Generic;
using DnsLib.ComponentLibrary.Macrolib;
using DnsLib.FactoryFloor.Utility;
using Nest;

namespace DnsLib.FactoryFloor.ElasticSearchComponent
{
    #region using directives

    #endregion

    /// <summary>The write and read back es tweet.</summary>
    public static class WriteAndDupEsTweet
    {
        /// <summary>The execute.</summary>
        /// <param name="pTweet">The p tweet.</param>
        /// <param name="client">The client.</param>
        /// <param name="esIndexName"></param>
        /// <returns>
        ///     The
        ///     <see>
        ///         <cref>List</cref>
        ///     </see>
        ///     .
        /// </returns>
        public static List<TweetModel> Execute(TweetModel pTweet, ElasticClient client,
            string esIndexName = "rtest.current")
        {
            var tweet = pTweet;
            var response = client.Index(tweet, idx => idx.Index(esIndexName));
            if (!response.ApiCall.Success) GeneralOperations.HandleBailingOut(Console.Out, "ApiCall.Success != true");

            var tweet2 = tweet;
            tweet2.Id = response.Id;
            return new List<TweetModel> {tweet, tweet2};
        }
    }
}
