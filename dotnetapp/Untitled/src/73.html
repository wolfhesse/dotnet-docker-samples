namespace DnsLib.FactoryFloor.Utility
{
    /// <summary>
    ///     The onetab operations options.
    /// </summary>
    public class OnetabOperationsOptions
    {
        /// <summary>Initializes a new instance of the <see cref="OnetabOperationsOptions" /> class.</summary>
        /// <param name="flgOut">The flg out.</param>
        /// <param name="json">The json.</param>
        public OnetabOperationsOptions(bool flgOut = true, string json = null)
        {
            FlgOut = flgOut;
            Json = json;
        }

        /// <summary>Gets a value indicating whether flg out.</summary>
        public bool FlgOut { get; }

        /// <summary>Gets or sets the json.</summary>
        public string Json { get; set; }
    }
}
