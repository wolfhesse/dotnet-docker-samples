using DnsLib.FactoryFloor.Utility;
using DnsLib.SysRes;
using Xunit;

namespace DnsLib.TestDriving.UserStories
{
    public class VirtualActorNodeInterfaceTest
    {
        [Fact(Timeout = 2500)]
        public static void EstablishTwoNodes()
        {
            var node1 = new ActorNodeBaseImpl(name: "node1 - EventSource, sender");
            var node2 = new ActorNodeBaseImpl(name: "node2 - EventHandler, processor");
            node1.RegisterAsEventHandler(node2);
            EnvironmentManager.WriteLine(node1, "dump");
            node1.ConnectionsList.ForEach(virtualActorNode =>
                EnvironmentManager.WriteLine(s: virtualActorNode, "dump"));
        }
    }
