using System;
using DnsLib.ComponentLibrary.Lab;
using DnsLib.FactoryFloor.TodoComponent;
using Xunit;


namespace DnsLib.TestDriving.Trainer
{
    #region using directives

    #endregion

    /// <summary>The test todo component.</summary>
    public class TestTodoComponent
    {
        /// <summary>The task added.</summary>
        private bool _taskAdded;

        /// <summary>The test add task.</summary>
        [Fact(Timeout = 2500)]
        public static void TestAddTask()
        {
            InMemoryTodoEngine.Init();

            var startCount = TodoController.TodoRepository.Count;

            TodoController.AddTodo("eins");
            TodoController.AddTodo("eins");

            Assert.True(startCount + 2 == TodoController.TodoRepository.Count);
        }

        /// <summary>The test todo engine add task.</summary>
        [Fact(Timeout = 2500)]

        // 
        public void TestTodoEngineAddTask()
        {
            // one way to do it: flags, local state
            _taskAdded = false;

            InMemoryTodoEngine.Init();
            var startCount = TodoController.TodoRepository.Count;
            TodoController.TodoRepository.EvTodoAdded += TaskRepositoryEvTodoAdded;

            TodoController.AddTodo("1");
            var todoItem = TodoController.TodoRepository.FindTodo(0);

            Assert.True(_taskAdded);
            Assert.True(0 == todoItem.Id);
            Assert.True(startCount + 1 == TodoController.TodoRepository.Count);
        }

        /// <summary>The task repository ev task added.</summary>
        /// <param name="sender">The sender.</param>
        /// <param name="e">The e.</param>
        private void TaskRepositoryEvTodoAdded(object sender, EventArgs e)
        {
            Console.Out.WriteLine($"{sender} added TodoItem");
            _taskAdded = true;
        }
    }
}
