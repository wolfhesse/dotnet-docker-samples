using System.Collections.Generic;
using System.Linq;
using DnsLib.FloorPlaan.PkgCt.EntityLib;
using DnsLib.FloorPlaan.PkgCt.InterfaceLib;

namespace DnsLib.FloorPlaan.PkgCt.CustomerProviderLib
{
    public class InMemoryCustomerProvider : ICustomerProvider
    {
        private readonly List<ICustomer> _customers = new List<ICustomer>();

        public void CreateCustomer(string name)
        {
            _customers.Add(new CtCustomer(name));
        }

        public int CustomerCount()
        {
            return _customers.Count;
        }

        public ICustomer GetCustomer(string name)
        {
            return _customers.Find(x => x.Name == name);
        }

        public List<ICustomer> Load()
        {
            return _customers;
        }


        public ICustomer Save(ICustomer customer)
        {
            var needInsert = true;
            var customerRet = customer;
            // upsert customer
            _customers.Where(x => x.Name == customer.Name).ToList()
                .ForEach(x =>
                {
                    customerRet = PatchCustomer(x, customer);
                    needInsert = false;
                });
            if (needInsert) _customers.Add(customer);

            return customerRet;
        }

        private ICustomer PatchCustomer(ICustomer customer, ICustomer arg)
        {
            customer.Name = arg.Name;
            customer.Description = arg.Description;
            return customer;
        }
    }
}
