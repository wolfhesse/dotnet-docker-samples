using DnsLib.FloorPlaan.PkgCt.InterfaceLib;
using Newtonsoft.Json;

namespace DnsLib.FloorPlaan.PkgCt.EntityLib
{
    public class CtCustomer : ICustomer
    {
        public CtCustomer()
        {
        }

        public CtCustomer(string name, string description = "n/a")
        {
            Name = name;
            Description = description;
        }

        [JsonProperty("description")] public string Description { get; set; }

        [JsonProperty("name")] public string Name { get; set; }

        public static ICustomer CustomerFrom((int id, string name, int balance) record)
        {
            return new CtCustomer
            {
                Name = record.name,
                Description = $@"balance is {record.balance}"
            };
        }

        public static ICustomer CustomerFrom((int id, string name, string description) record2)
        {
            return new CtCustomer {Description = record2.description, Name = record2.name};
        }
    }
}
