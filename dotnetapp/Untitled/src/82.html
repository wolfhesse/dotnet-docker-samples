using System;

namespace DnsLib.SysRes
{
    public sealed class FactoryFloorController
    {
        /*
         * FactoryFloor initialized Online
         */
        private static bool _stageOnline;
        public event EventHandler EvStageOffline;
        public event EventHandler EvStageOnline;

        public static string CurrentStageOnlineCondition =>
            "FactoryFloorController: STAGE == " + (_stageOnline ? "ONLINE" : "OFFLINE");

        public static FactoryFloorController Instance { get; } = new FactoryFloorController();

        public static void FnTweetMessage(string message)
        {
            try
            {
                var t = TweetEngine.BuildTweet("rogera");
                t.Value = message;
                var res = EsOperationsEngine.EsWriteAndDupTweet(t);
                EnvironmentManager.WriteTweetToConsole(res[0]);
            }
            catch
            {
                SetStageOffline();
            }
        }


        public static bool IsStageOnline()
        {
            return _stageOnline;
        }

        public static void SetStageOffline()
        {
            _stageOnline = false;
            Instance?.OnEvStageOffline();
        }

        public static void SetStageOnline()
        {
            _stageOnline = true;
            Instance?.OnEvStageOnline();
        }

        private void OnEvStageOffline()
        {
            EvStageOffline?.Invoke(this, EventArgs.Empty);
        }

        private void OnEvStageOnline()
        {
            EvStageOnline?.Invoke(this, EventArgs.Empty);
        }
    }
}
