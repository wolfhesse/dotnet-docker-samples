        /// <summary>The coders resource tuple parsing.</summary>
        [Fact(Timeout = 2500)]
        public static void CodersResourceTupleParsing()
        {
            var programmerResourcesList = CodersResourceCollection.GetProgrammerResourcesList();
            programmerResourcesList.ForEach(
                s =>
                {
                    // Console.Out.WriteLine(@"k = " + resourceKey);
                    CodersResourceCollection.GetElement(s).ForEach(
                        possibleTuple => { CodersResourceCollection.ParseForPossibleTuple(possibleTuple, s); });
                });
        }
