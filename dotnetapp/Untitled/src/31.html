using System;

namespace DnsLib.FactoryFloor.TodoComponent
{
    #region using directives

    #endregion

    /// <summary>The task item.</summary>
    public class TodoItem
    {
        /// <summary>Initializes a new instance of the <see cref="TodoItem" /> class.</summary>
        /// <param name="title">The title.</param>
        public TodoItem(string title)
        {
            CreatedAt = DateTimeOffset.Now;
            Title = title;
        }

        /// <summary>Initializes a new instance of the <see cref="TodoItem" /> class.</summary>
        internal TodoItem()
        {
        }

        /// <summary>Gets or sets the created at.</summary>
        public DateTimeOffset CreatedAt { get; set; }

        /// <summary>Gets or sets the extra creator info.</summary>
        public Type ExtraCreatorInfo { get; set; }

        /// <summary>Gets or sets the id.</summary>
        public int Id { get; set; }

        /// <summary>Gets or sets the title.</summary>
        public string Title { get; set; }

        /// <summary>The is valid.</summary>
        /// <returns>The <see cref="bool" />.</returns>
        public bool IsValid()
        {
            return Title.Trim().Length > 0;
        }
    }
}
