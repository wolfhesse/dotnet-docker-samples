using System.Collections.Generic;
using System.Diagnostics;
using DnsLib.FactoryFloor.TodoComponent;

namespace DnsLib.ComponentLibrary.Lab
{
    #region using directives

    #endregion

    /// <inheritdoc />
    /// <summary>The in memory task repository.</summary>
    public class InMemoryTodoRepository : AbstractTodoRepository
    {
        /// <summary>The tasks.</summary>
        private readonly List<TodoItem> _todoItems;

        /// <summary>The id.</summary>
        private int _id;

        /// <summary>Initializes a new instance of the <see cref="InMemoryTodoRepository" /> class.</summary>
        public InMemoryTodoRepository()
        {
            _todoItems = new List<TodoItem>();
            _id = 0;
        }

        /// <summary>The count.</summary>
        public override int Count => _todoItems.Count;

        /// <summary>The find task.</summary>
        /// <param name="taskId">The task id.</param>
        /// <returns>The <see cref="TodoItem" />.</returns>
        public override TodoItem FindTodo(int taskId)
        {
            Debug.Assert(_todoItems != null, "_todoItems != null");
            return _todoItems[taskId];
        }

        /// <summary>The persist.</summary>
        /// <param name="t">The t.</param>
        public override void Persist(TodoItem t)
        {
            t.Id = _id++;
            _todoItems.Add(t);
            OnEvTaskAdded(t);
        }
    }
}
