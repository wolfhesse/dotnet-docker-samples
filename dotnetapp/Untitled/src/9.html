using System;
using System.IO;
using DnsLib.AbstractArchitecture.AseArchitecture.ProgramSetup;
using DnsLib.FactoryFloor.Utility;

namespace DnsLib.TestDriving.Trainer
{
    #region using directives

    #endregion

    /// <summary>The dates container operations.</summary>
    public static class DatesContainerOperations
    {
        /// <summary>Gets or sets the port.</summary>
        public static PresenterBase Port { get; set; }
    }

    /// <summary>The controller.</summary>
    public class Controller
    {
        /// <summary>The engine setup.</summary>
        private readonly EngineSetup _engineSetup;

        /// <summary>Initializes a new instance of the <see cref="Controller" /> class.</summary>
        /// <param name="engineSetup">The engine setup.</param>
        /// <param name="textWriter">The text writer.</param>
        public Controller(EngineSetup engineSetup, TextWriter textWriter)
        {
            DatesContainerOperations.Port = new PresenterBase {TextWriter = textWriter};
            DatesContainerOperations.Port.TextWriter.WriteLine("controller initialization...");

            _engineSetup = engineSetup;

            DatesContainerOperations.Port.TextWriter.WriteLine("controller initialized...");
        }

        /// <summary>The after action.</summary>
        public event EventHandler AfterAction;

        /// <summary>The action batch.</summary>
        public void ActionBatch()
        {
            _engineSetup.DatesContainerModel.CreateDates();
            _engineSetup.DatesContainerModel.InsertTimestamp();
            _engineSetup.DatesContainerModel.Sort();
            OnAfterAction();
        }

        /// <summary>The on after action.</summary>
        protected virtual void OnAfterAction()
        {
            AfterAction?.Invoke(this, EventArgs.Empty);
        }
    }
}
