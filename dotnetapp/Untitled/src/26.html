using System;
using System.ComponentModel;

namespace DnsLib.FactoryFloor.Utility
{
    #region using directives

    #endregion

    /// <summary>The component 1.</summary>
    public class Component1 : Component
    {
        /// <summary>The standard engine one.</summary>
        private IStandardEngineOne _standardEngineOne;

        /// <summary>
        ///     Required designer variable.
        /// </summary>
        // ReSharper disable once NotAccessedField.Local
        private IContainer components;

        /// <summary>Initializes a new instance of the <see cref="Component1" /> class.</summary>
        public Component1()
        {
            InitializeComponent();
        }

        /// <summary>Initializes a new instance of the <see cref="Component1" /> class.</summary>
        /// <param name="container">The container.</param>
        public Component1(IContainer container)
        {
            container.Add(this);

            InitializeComponent();
        }

        /// <summary>The ev processing.</summary>
        public event EventHandler EvProcessing;

        /// <summary>The ev started.</summary>
        public event EventHandler EvStarted;

        /// <summary>The ev stopped.</summary>
        public event EventHandler EvStopped;

        /// <summary>The start.</summary>
        /// <param name="standardEngineOne">The standard engine one.</param>
        public void Start(IStandardEngineOne standardEngineOne)
        {
            _standardEngineOne = standardEngineOne;
            ((StandardEngineOne) standardEngineOne).EvAfterFinalizing += Component1EvAfterFinalizing;
            OnEvStarted();

            //// processing
            Process();
            Stop();
        }

        /// <summary>The stop.</summary>
        public void Stop()
        {
            OnEvStopped();
        }

        /// <summary>The on ev processing.</summary>
        protected virtual void OnEvProcessing()
        {
            EvProcessing?.Invoke(this, EventArgs.Empty);
        }

        /// <summary>The on ev started.</summary>
        protected virtual void OnEvStarted()
        {
            EvStarted?.Invoke(this, EventArgs.Empty);
        }

        /// <summary>The on ev stopped.</summary>
        protected virtual void OnEvStopped()
        {
            EvStopped?.Invoke(this, EventArgs.Empty);
        }

        /// <summary>The component 1 ev after finalizing.</summary>
        /// <param name="sender">The sender.</param>
        /// <param name="e">The e.</param>
        private void Component1EvAfterFinalizing(object sender, EventArgs e)
        {
            Console.Out.WriteLine("1sender = {0}", sender);
        }

        /// <summary>
        ///     Required method for Designer support - do not modify
        ///     the contents of this method with the code editor.
        /// </summary>
        private void InitializeComponent()
        {
            components = new Container();
        }

        /// <summary>The process.</summary>
        private void Process()
        {
            OnEvProcessing();
            _standardEngineOne.Start();
        }
    }
}
