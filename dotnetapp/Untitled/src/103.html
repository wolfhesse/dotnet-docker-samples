using System;
using System.Collections.Generic;
using DnsLib.FloorPlaan.PkgCt.InterfaceLib;
using Newtonsoft.Json;

namespace DnsLib.FloorPlaan.PkgCt.BusinessPackageCustomer
{
    public class CustomerRepository : ICustomerRepository
    {
        private static ICustomerProvider _customerProvider;

        public CustomerRepository(ICustomerProvider customerProvider)
        {
            _customerProvider = customerProvider;
        }

        public void CreateCustomer(string name)
        {
            _customerProvider.CreateCustomer(name);
        }

        public int CustomerCount()
        {
            return _customerProvider.CustomerCount();
        }

        public void DumpCustomers()
        {
            var customers = _customerProvider.Load();
            Console.Out.WriteLine("customers = {0}",
                JsonConvert.SerializeObject(
                    customers, Formatting.Indented));
        }

        public ICustomer GetCustomer(string name)
        {
            return _customerProvider.GetCustomer(name);
        }

        public List<ICustomer> GetCustomers()
        {
            return _customerProvider.Load();
        }

        public ICustomer Save(ICustomer customer)
        {
            return
                _customerProvider.Save(customer);
        }
    }
}
