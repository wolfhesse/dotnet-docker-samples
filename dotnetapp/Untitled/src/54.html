namespace DnsLib.FactoryFloor.Utility
{
    /// <summary>The engine setup.</summary>
    public class EngineSetup
    {
        /// <summary>The dates container model.</summary>
        public readonly DatesContainerModel DatesContainerModel;

        /// <summary>The view.</summary>
        // ReSharper disable once NotAccessedField.Local
        private readonly IView _view;

        /// <summary>Initializes a new instance of the <see cref="EngineSetup" /> class.</summary>
        /// <param name="datesContainerModel">The dates container model.</param>
        /// <param name="view">The view.</param>
        public EngineSetup(DatesContainerModel datesContainerModel, IView view)
        {
            DatesContainerModel = datesContainerModel;
            _view = view;
        }
    }
}
