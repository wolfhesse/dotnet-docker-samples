using System;
using System.Collections.Generic;

namespace DnsLib.AbstractArchitecture.AseArchitecture.ProgramSetup
{
    /// <summary>The nvram ref attribute.</summary>
    public sealed class NvramRefAttribute : Attribute
    {
        /// <summary>The stage.</summary>
        private Stage _stage;

        /// <summary>Initializes a new instance of the <see cref="NvramRefAttribute" /> class.</summary>
        /// <param name="stage">The stage.</param>
        public NvramRefAttribute(Stage stage)
        {
            _stage = stage;
        }

        /// <summary>The stage.</summary>
        public enum Stage
        {
            A1,
            A2,
            A3,
            A4,

            /// <summary>The candidate.</summary>
            Candidate,
            Contender,
            President
        }

        public List<string> AddStage()
        {
            var messages = new List<string>();
            var stageText = $"{_stage} [w/addition]";
            messages.Add(stageText);
            switch (_stage)
            {
                case Stage.Candidate:
                    _stage = Stage.Contender;
                    break;
                case Stage.Contender:
                    messages.Add("E NO NEXT STAGE");
                    break;
                default:
                {
                    messages.Add("SHOULD NOT HAPPEN");
                    throw new NotImplementedException();
                }
            }

            return messages;
        }
    }
}
