using System;
using HeyRed.MarkdownSharp;
using ReverseMarkdown;

namespace DnsLib.FloorPlaan.PkgCt.NetworkAdapter
{
    public static class MarkdownExtension
    {
        public static string ConcatenateToMarkdown(string html, string markdown)
        {
            var md = new Markdown(new MarkdownOptions
            {
                AutoNewLines = true,
                DisableHr = true,
                DisableHeaders = false
            });
            var converter = new Converter(new Config(removeComments: true));

            return converter.Convert(
                md.Transform(
                    converter.Convert(html) + Environment.NewLine + markdown));
        }
    }
}
