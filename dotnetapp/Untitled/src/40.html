using System;
using Newtonsoft.Json;

namespace DnsLib.FactoryFloor.TodoComponent
{
    #region using directives

    #endregion

    /// <summary>The task event args.</summary>
    public class TodoEventArgs : EventArgs
    {
        /// <summary>The task item.</summary>
        private readonly TodoItem _todoItem;

        /// <summary>Initializes a new instance of the <see cref="TodoEventArgs" /> class.</summary>
        /// <param name="todoItem">The task item.</param>
        public TodoEventArgs(TodoItem todoItem)
        {
            _todoItem = todoItem;
        }

        /// <summary>The to string.</summary>
        /// <returns>The <see cref="string" />.</returns>
        public override string ToString()
        {
            return base.ToString() + Environment.NewLine
                                   + JsonConvert.SerializeObject(_todoItem, Formatting.Indented);
        }
    }
}
