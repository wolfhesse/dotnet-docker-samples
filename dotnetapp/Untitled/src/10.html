using System;
using DnsLib.FactoryFloor.Utility;
using DnsLib.SysRes;
using Newtonsoft.Json;

namespace DnsLib.ComponentLibrary.Lab
{
    #region using directives

    #endregion

    /// <summary>The onetab exper engine.</summary>
    public static class OnetabExperEngine
    {
        /// <summary>The fn run json onetab exper.</summary>
        /// <param name="onetabOperationsOptions">The onetab operations options.</param>
        /// <returns>The <see cref="string" />.</returns>
        public static string FnRunJsonOnetabExper(OnetabOperationsOptions onetabOperationsOptions)
        {
            if (null == onetabOperationsOptions.Json)
                onetabOperationsOptions.Json =
                    SampleDataProvider.OnetabJsonEmptyState;
            var deserializeObject = JsonConvert.DeserializeObject(onetabOperationsOptions.Json);

            if (onetabOperationsOptions.FlgOut) Console.Out.WriteLine(deserializeObject);

            var back = JsonConvert.SerializeObject(deserializeObject);
            if (onetabOperationsOptions.FlgOut) Console.Out.WriteLine(back);
            return back;
        }

        /// <summary>The write data f.</summary>
        public static void WriteDataF()
        {
            var refText = typeof(QueueControllerExperEngine) + ".WriteDataF";
            var sw = EnvironmentManager.GetResultStreamWriter();
            sw.WriteLine(new string('=', 72));
            sw.WriteLine(refText);
            sw.WriteLine("\trunning @ " + DateTimeOffset.Now.LocalDateTime);
            sw.Close();
        }
    }
}
