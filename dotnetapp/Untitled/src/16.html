using DnsLib.ComponentLibrary.Authentication;
using DnsLib.SysRes;
using WooCommerceNET;

namespace DnsLib.FactoryFloor.ShopComponent
{
    /// <summary>The woo stuff auth adapter.</summary>
    public static class WooStuffAuthAdapter
    {
        /// <summary>The fn auth for rcs 1.</summary>
        /// <returns>The <see cref="AuthContainer" />.</returns>
        public static AuthContainer FnAuthForRcs1()
        {
            var auth = new AuthContainer();
            AseAuthVaultForTesting.PutRcs1ConsumerAuthInto(auth);
            return auth;
        }

        /// <summary>The fn auth for rcs 2.</summary>
        /// <returns>The <see cref="AuthContainer" />.</returns>
        public static AuthContainer FnAuthForRcs2()
        {
            var auth = new AuthContainer();
            AseAuthVaultForTesting.PutRcs2ConsumerAuthInto(auth);
            return auth;
        }

        /// <summary>The fn auth for test apo r 1.</summary>
        /// <returns>The <see cref="AuthContainer" />.</returns>
        public static AuthContainer FnAuthForRcs2Wrong()
        {
            var auth = new AuthContainer();
            AseAuthVaultForTesting.PutRcs2WrongConsumerAuthInto(auth);
            return auth;
        }

        /// <summary>The fn rest api rcs 1.</summary>
        /// <returns>The <see cref="RestAPI" />.</returns>
        public static RestAPI FnRestApiRcs1()
        {
            var auth = FnAuthForRcs1();
            var restApi = new RestAPI(PlatformSysGen.Rcs1RestApiEndpoint, $"{auth.Key}", $"{auth.Secret}");
            return restApi;
        }

        public static RestAPI FnRestApiRcs2()
        {
            var auth = FnAuthForRcs2();
            var restApi = new RestAPI(PlatformSysGen.Rcs2RestApiEndpoint, $"{auth.Key}", $"{auth.Secret}");
            return restApi;
        }

        /// <summary>The rest api default.</summary>
        /// <param name="restApiEndpoint">The rest api endpoint.</param>
        /// <param name="auth">The auth.</param>
        /// <returns>The <see cref="RestAPI" />.</returns>
        public static RestAPI RestApiDefault(string restApiEndpoint = null, AuthContainer auth = null)
        {
            restApiEndpoint = restApiEndpoint ?? PlatformSysGen.Rcs2RestApiEndpoint;
            auth = auth ?? FnAuthForRcs2();

            return new RestAPI(restApiEndpoint, $"{auth.Key}", $"{auth.Secret}");
        }
    }
}
