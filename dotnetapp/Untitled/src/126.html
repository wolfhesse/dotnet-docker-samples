using System.Collections.Generic;
using DnsLib.FloorPlaan.PkgCt.EntityLib;
using Newtonsoft.Json;

namespace DnsLib.FloorPlaan.PkgCt.TypeUtilities
{
    public class DataContainerFuerAustauschCustomerAccount
    {
        [JsonProperty("accounts")] public List<CtAccount> Accounts = new List<CtAccount>();
        [JsonProperty("balance")] public int Balance;
        [JsonProperty("customerName")] public string CustomerName;

        public void ComputeBalance()
        {
            Accounts.ForEach(x => { Balance += x.Balance; });
        }
    }
}
