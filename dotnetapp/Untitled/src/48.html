using System;
using DnsLib.FactoryFloor.ShopComponent.AseWooCommerceNET;

namespace DnsLib.FactoryFloor.Utility
{
    public class MachineOne
    {
        private static bool _onlineState;

        // ReSharper disable once NotAccessedField.Local
        private WooCommerceAdapter _wooCommerceAdapter;

        public MachineOne(WooCommerceAdapter wooCommerceAdapter)
        {
            Console.WriteLine(this);
            _wooCommerceAdapter = wooCommerceAdapter;
        }

        public event EventHandler EvSystemOnline;

        public bool OnlineState
        {
            get => _onlineState;
            set => _onlineState = value;
        }

        public static bool TakeSystemOnline()
        {
            _onlineState = true;

            return _onlineState;
        }

        public bool CheckOnlineState()
        {
            if (_onlineState) OnEvSystemOnline();

            return _onlineState;
        }

        protected virtual void OnEvSystemOnline()
        {
            EvSystemOnline?.Invoke(this, EventArgs.Empty);
        }
    }
}
