using System;
using System.Threading.Tasks;
using DnsLib.SysRes;
using WooCommerceNET;
using WooCommerceNET.WooCommerce.v2;

namespace DnsLib.FactoryFloor.ShopComponent.AseWooCommerceNET
{
    #region using directives

    #endregion

    /// <summary>The woo commerce adapter.</summary>
    public class WooCommerceAdapter : IShopAdapter
    {
        /// <summary>The rest api.</summary>
        private RestAPI _restApi;

        /// <summary>The add product.</summary>
        /// <param name="product">The product.</param>
        /// <returns>The <see cref="Product" />.</returns>
        public Product AddProduct(Product product)
        {
            var s = InternalAddProductTask(_restApi, product).Result;
            return s;
        }

        /// <summary>The set rest api.</summary>
        /// <param name="restApi">The rest api.</param>
        public void SetRestApi(RestAPI restApi)
        {
            _restApi = restApi;
        }

        /// <summary>The internal add product task.</summary>
        /// <param name="restApi">The rest api.</param>
        /// <param name="p">The p.</param>
        /// <returns>The <see cref="Task" />.</returns>
        private static async Task<Product> InternalAddProductTask(RestAPI restApi, Product p)
        {
            var resProduct = new Product
            {
                name = "software-internal: unprocessed product created;"
            };

            // create unique TaskId
            var taskId = Guid.NewGuid();

            // var start = DateTime.Now;
            var wc = new WCObject(restApi);
            Console.WriteLine($@"		{typeof(WooCommerceAdapter)} before wc.Product.Add
	TaskId: {taskId}");
            try
            {
                resProduct = await wc.Product.Add(p);
                Console.WriteLine($@"product created {resProduct.name}");
            }
            catch (TimeoutException timeout)
            {
                EnvironmentManager.WriteLine("timeout: " + timeout);
            }
            catch (Exception ex)
            {
                EnvironmentManager.WriteLine("general exception: " + ex);
            }

            Console.WriteLine($@"		{typeof(WooCommerceAdapter)} after wc.Product.Add
	TaskId: {taskId}");

            // var duration = DateTime.Now - start;
            // Console.WriteLine($"duration to add product: {duration}");
            return resProduct;
        }
    }
}
