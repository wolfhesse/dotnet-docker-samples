using System;
using System.Diagnostics.CodeAnalysis;
using DnsLib.FactoryFloor.Utility;
using Xunit;


namespace DnsLib.TestDriving.Trainer
{
    #region using directives

    #endregion

    /// <summary>
    ///     The engine one test.
    /// </summary>
    [SuppressMessage(
        "StyleCop.CSharp.MaintainabilityRules",
        "SA1402:FileMayOnlyContainASingleClass",
        Justification = "Reviewed. Suppression is OK here.")]
    public static class EngineSampleTest
    {
        class Handling
        {
            /// <summary>The handle env fin.</summary>
            /// <param name="sender">The sender.</param>
            /// <param name="e">The e.</param>
            public static void HandleEnvAfterFinalizing(object sender, EventArgs e)
            {
                Console.Out.WriteLine("sender = {0}", sender);
                ((IStandardEngineOne) sender).Result = true;
            }
        }

        /// <summary>
        ///     The generalized E-Procedure test.
        /// </summary>
        [Fact(Timeout = 2500)]
        // 
        public static void GeneralizedEngineSampleTest()
        {
            const bool expected = true;
            var engineSample = new StandardEngineOne();

            engineSample.EvAfterFinalizing += Handling.HandleEnvAfterFinalizing;
            Console.Out.WriteLine("standardEngineOne = {0}", engineSample);
            engineSample.Start();
            Console.Out.WriteLine("standardEngineOne = {0}", engineSample);
            Assert.Equal(expected, StandardEngineOne.Result);
        }
    }
}
