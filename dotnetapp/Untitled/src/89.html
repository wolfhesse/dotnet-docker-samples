using DnsLib.AbstractArchitecture.AseArchitecture;
using Xunit;


namespace DnsLib.TestDriving.Core
{
    /// <summary>The formal ase dynamic stages tests.</summary>
    public class FormalAseDynamicStagesTests
    {
        public Dynamics.ProcessingStage Stage { get; set; }

        /// <summary>The processing stages test.</summary>
        [Fact(Timeout = 2500)]
        public void ProcessingStagesTest()
        {
            Stage = Dynamics.ProcessingStage.Abended;
            Stage = Dynamics.ProcessingStage.Abnormed;
            Stage = Dynamics.ProcessingStage.Cancelled;
            Stage = Dynamics.ProcessingStage.Idle;
            Stage = Dynamics.ProcessingStage.InflightProcessing;
            Stage = Dynamics.ProcessingStage.InflightPublishing;
            Stage = Dynamics.ProcessingStage.InflightReading;
            Stage = Dynamics.ProcessingStage.Waiting;

            Assert.True(8 == (int) Dynamics.ProcessingStage.LastInLine);
        }
    }
}
