using System;
using System.ComponentModel;
using DnsLib.FactoryFloor.Utility;

namespace DnsLib.ComponentLibrary.Lab
{
    #region using directives

    #endregion

    /// <inheritdoc />
    /// <summary>
    ///     The process layer one.
    /// </summary>
    public class ProcessLayerSample : Component
    {
        /// <summary>The component 11.</summary>
        private Component1 _component11;

        /// <summary>The component 21.</summary>
        private Component2 _component21;

        /// <summary>
        ///     Required designer variable.
        /// </summary>
        private IContainer components;

        /// <summary>Initializes a new instance of the <see cref="ProcessLayerSample" /> class.</summary>
        public ProcessLayerSample()
        {
            InitializeComponent();
        }

        /// <summary>Initializes a new instance of the <see cref="ProcessLayerSample" /> class.</summary>
        /// <param name="container">The container.</param>
        public ProcessLayerSample(IContainer container)
        {
            container.Add(this);

            InitializeComponent();
        }

        /// <summary>Gets or sets the standard engine one.</summary>
        public IStandardEngineOne StandardEngineOne { get; set; }

        /// <summary>
        ///     The start.
        ///     External Event Triggers
        ///     EnvironmentObserver has configured handler
        /// </summary>
        /// <param name="one"></param>
        public void Start(IStandardEngineOne one = null)
        {
            if (StandardEngineOne == null)
                if (one != null)

                    // in constructor: lazy setup of components
                    StandardEngineOne = one;

            Console.Out.WriteLine("standardEngineOne = {0}", StandardEngineOne);
            _component11.Start(StandardEngineOne);
        }

        /// <summary>Clean up any resources being used.</summary>
        /// <param name="disposing">true if managed resources should be disposed; otherwise, false.</param>
        protected override void Dispose(bool disposing)
        {
            if (disposing && components != null) components.Dispose();
            base.Dispose(disposing);
        }

        /// <summary>The component 11 ev processing.</summary>
        /// <param name="sender">The sender.</param>
        /// <param name="e">The e.</param>
        private void Component11EvProcessing(object sender, EventArgs e)
        {
            Console.Out.WriteLine("e = {0}", e);
            Console.Out.WriteLine("s = {0}", sender);
        }

        /// <summary>The component 11 ev started.</summary>
        /// <param name="sender">The sender.</param>
        /// <param name="e">The e.</param>
        private void Component11EvStarted(object sender, EventArgs e)
        {
            Console.Out.WriteLine("e = {0}", e);
            Console.Out.WriteLine("s = {0}", sender);
        }

        /// <summary>The component 11 ev stopped.</summary>
        /// <param name="sender">The sender.</param>
        /// <param name="e">The e.</param>
        private void Component11EvStopped(object sender, EventArgs e)
        {
            Console.Out.WriteLine("e = {0}", e);
            Console.Out.WriteLine("s = {0}", sender);

            // schedule component 2 processing
            _component21.Start();
        }

        /// <summary>The component 21 ev processing.</summary>
        /// <param name="sender">The sender.</param>
        /// <param name="e">The e.</param>
        private void Component21EvProcessing(object sender, EventArgs e)
        {
            Console.Out.WriteLine("e = {0}", e);
            Console.Out.WriteLine("s = {0}", sender);
        }

        /// <summary>The component 21 ev started.</summary>
        /// <param name="sender">The sender.</param>
        /// <param name="e">The e.</param>
        private void Component21EvStarted(object sender, EventArgs e)
        {
            Console.Out.WriteLine("e = {0}", e);
            Console.Out.WriteLine("s = {0}", sender);
        }

        /// <summary>The component 21 ev stopped.</summary>
        /// <param name="sender">The sender.</param>
        /// <param name="e">The e.</param>
        private void Component21EvStopped(object sender, EventArgs e)
        {
            Console.Out.WriteLine("e = {0}", e);
            Console.Out.WriteLine("s = {0}", sender);

            Console.Out.WriteLine("standardEngineOne = {0}", StandardEngineOne);
        }

        /// <summary>
        ///     Required method for Designer support - do not modify
        ///     the contents of this method with the code editor.
        /// </summary>
        private void InitializeComponent()
        {
            components = new Container();
            _component11 = new Component1(components);
            _component21 = new Component2(components);
            // 
            // _component11
            // 
            _component11.EvProcessing += Component11EvProcessing;
            _component11.EvStarted += Component11EvStarted;
            _component11.EvStopped += Component11EvStopped;
            // 
            // _component21
            // 
            _component21.EvProcessing += Component21EvProcessing;
            _component21.EvStarted += Component21EvStarted;
            _component21.EvStopped += Component21EvStopped;
        }
    }
}
