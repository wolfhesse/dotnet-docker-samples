using System;

namespace DnsLib.FactoryFloor.TodoComponent
{
    #region using directives

    #endregion

    /// <summary>
    ///     The todo controller. component
    /// </summary>
    public abstract class TodoController
    {
        /// <summary>
        ///     Gets the task repository.
        /// </summary>
        public static AbstractTodoRepository TodoRepository { get; protected set; }

        /// <summary>The add task.</summary>
        /// <param name="title">The title.</param>
        public static void AddTodo(string title)
        {
            Console.Out.WriteLine(typeof(TodoController) + ".AddTodo");
            TodoComponent.AddTodo.Execute(title);
        }
    }
}
