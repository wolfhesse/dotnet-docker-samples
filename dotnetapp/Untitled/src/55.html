using System;
using System.Collections.Generic;
using DnsLib.FactoryFloor.ShopComponent;
using DnsLib.FactoryFloor.ShopComponent.AseWooCommerceNET;
using DnsLib.SysRes;
using WooCommerceNET;
using WooCommerceNET.WooCommerce.v2;

namespace DnsLib.TestDriving.Operations
{
    #region using directives

    #endregion

    /// <summary>The create products for test.</summary>
    public class CreateProductsForTest
    {
        /// <summary>The execute.</summary>
        /// <param name="restApi">The rest api.</param>
        /// <param name="productsToInsert"></param>
        public static void Execute(RestAPI restApi, List<Product> productsToInsert)
        {
            IShopAdapter adapter = new WooCommerceAdapter();
            IShopConfiguration configuration = new WooCommerceConfiguration(restApi);

            productsToInsert.ForEach(
                product => Console.Out.WriteLine(
                    $"{DateTimeOffset.Now}/{DateTimeOffset.Now.LocalDateTime.Millisecond} -> Product: {new ShopEngine(adapter, configuration).AddProduct(product).name}"));
        }
    }
}
