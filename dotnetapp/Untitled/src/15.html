using System;
using System.Collections.Generic;
using System.Diagnostics.CodeAnalysis;
using DnsLib.SysRes;
using Xunit;


namespace DnsLib.TestDriving.Core
{
    #region using directives

    #endregion

    /// <summary>
    ///     The environment manager tests.
    /// </summary>
    public class EnvironmentManagerTests
    {
        [Fact(Timeout = 2500)]
        public static void CallEnvironmentManagerFun()
        {
            var environmentManager = new EnvironmentManager();
            Console.Out.WriteLine("environmentManager = {0}", environmentManager);
            Console.Out.WriteLine("environmentManager.Version = {0}", environmentManager.Version);
        }

        /// <summary>
        ///     The get ase data d test.
        /// </summary>
        [SuppressMessage(
            "StyleCop.CSharp.DocumentationRules",
            "SA1650:ElementDocumentationMustBeSpelledCorrectly",
            Justification = "Reviewed. Suppression is OK here.")]
        [Fact(Timeout = 2500)]
        public static void GetAseDataDTest()
        {
            Assert.Equal(EnvironmentManager.AseDataD, EnvironmentManager.GetDataD());
        }

        /// <summary>
        ///     The get configuration.
        /// </summary>
        [Fact(Timeout = 2500)]
        public void GetConfigurationTest()
        {
            // object GetConfiguration(string key)
            IDictionary<string, object> dictionary = new Dictionary<string, object> {["eins"] = 1, ["zwo"] = 2};

            // return dictionary[key];
            var expected = dictionary["eins"];
            var configuration = EnvironmentManager.Configuration;
            var actual = configuration["eins"];
            Assert.Equal(expected, actual);
        }
    }
}
