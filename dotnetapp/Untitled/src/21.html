namespace DnsLib.FactoryFloor.TodoComponent
{
    /// <summary>The TodoRepository interface.</summary>
    public abstract class AbstractTodoRepository
    {
        /// <summary>The task added event handler.</summary>
        /// <param name="sender">The sender.</param>
        /// <param name="e">The e.</param>
        public delegate void TodoAddedEventHandler(object sender, TodoEventArgs e);

        /// <summary>The ev task added.</summary>
        public event TodoAddedEventHandler EvTodoAdded;

        /// <summary>Gets the count.</summary>
        public abstract int Count { get; }

        /// <summary>The find task.</summary>
        /// <param name="taskId">The task id.</param>
        /// <returns>The <see cref="TodoItem" />.</returns>
        public abstract TodoItem FindTodo(int taskId);

        /// <summary>The persist.</summary>
        /// <param name="t">The t.</param>
        public abstract void Persist(TodoItem t);

        /// <summary>The on ev task added.</summary>
        /// <param name="todoItem">The task item.</param>
        protected void OnEvTaskAdded(TodoItem todoItem)
        {
            var eventArgs = new TodoEventArgs(todoItem);
            EvTodoAdded?.Invoke(this, eventArgs);
        }
    }
}
